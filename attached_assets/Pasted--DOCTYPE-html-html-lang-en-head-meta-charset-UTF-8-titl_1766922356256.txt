<script>
// ---------------- SYSTEM CORE ----------------
const hexStats={STR:50,INT:50,SPI:50,VIT:50,WIS:50,DIS:50};
let FTP=0;
let completed=new Set();

function log(msg){
  const l=document.getElementById("log");
  l.innerHTML+=`<div>[SYSTEM] ${msg}</div>`;
  l.scrollTop=l.scrollHeight;
}

// ---------------- HEX CHART ----------------
const chart=new Chart(document.getElementById("hexChart"),{
  type:"radar",
  data:{
    labels:Object.keys(hexStats),
    datasets:[{
      data:Object.values(hexStats),
      borderColor:"#FF1D1D",
      backgroundColor:"rgba(255,29,29,.3)",
      pointBackgroundColor:"#FFF"
    }]
  },
  options:{scales:{r:{min:0,max:100}},plugins:{legend:{display:false}}}
});

function setHex(stat,val){
  hexStats[stat]=Number(val);
  chart.data.datasets[0].data=Object.values(hexStats);
  chart.update();
  updateLevel();
  log(`${stat} recalibrated → ${val}`);
}

function updateLevel(){
  const lvl=Math.floor(Object.values(hexStats).reduce((a,b)=>a+b,0)/60);
  document.getElementById("level").innerText=lvl;
}

// ---------------- QUESTS ----------------
const fixedTasks=[
  "Deep Work (7 Flows)",
  "Physical Training",
  "Eat 4 Meals",
  "Read 5 Chapters",
  "Spiritual Bread"
];

const randomPool=[
  "Clean Workspace",
  "Engineering Flashcards",
  "Meditation / Prayer",
  "Educational Podcast",
  "Drink 3L Water"
];

function render(){
  const f=document.getElementById("fixed");
  const r=document.getElementById("random");
  f.innerHTML=""; r.innerHTML="";
  fixedTasks.forEach(t=>{
    f.innerHTML+=`<li><input type="checkbox" onchange="toggle('${t}')"> ${t}</li>`;
  });
  shuffle([...randomPool]).slice(0,3).forEach(t=>{
    r.innerHTML+=`<li><input type="checkbox" onchange="toggle('${t}')"> ${t}</li>`;
  });
}

function toggle(t){
  completed.has(t)?completed.delete(t):completed.add(t);
}

function claim(){
  if(completed.size<5){
    log("Reward denied. Quota not met.");
    return;
  }
  FTP+=10;
  document.getElementById("ftp").innerText=FTP;
  completed.clear();
  render();
  log("Daily Complete → +10 FTP");
}

function shuffle(a){
  for(let i=a.length-1;i>0;i--){
    const j=Math.floor(Math.random()*(i+1));
    [a[i],a[j]]=[a[j],a[i]];
  }
  return a;
}

render();
log("System Initialized.");
</script>
